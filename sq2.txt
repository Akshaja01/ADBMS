CREATE TABLE DEPT(DEPTID INT,NAME VARCHAR(80),CONSTRAINT Q PRIMARY KEY(DEPTID));
CREATE TABLE STUDNT(STDID INT,NAME VARCHAR(80),CLASS VARCHAR(60),CITY VARCHAR(10),TOTAL_MARKS INT,PERCENTAGE NUMBERIC(5),DEPTID INT,CONSTRAINT pKEY PRIMARY KEY(STDID),CONSTRAINT DEL FOREIGN KEY(DEPTID) REFERENCES DEPT(DEPTID));
CREATE TABLE MARK(ROLLNO INT,SUB1 INT,SUB2 INT,SUB3 INT,CONSTRAINT MARK FOREIGN KEY(ROLLNO) REFERENCES STUDEN(ROLL_NO));
CREATE TABLE STAFF(STAFFID INT,NAME VARCHAR(70),DEPTID INT,DESIGNATION VARCHAR(15),SALARY NUMERIC(8),CITY VARCHAR(80),CONSTRAINT STF PRIMARY KEY(STAFFID),CONSTRAINT DEPT FOREIGN KEY(DEPTID) REFERENCES DEPT(DEPTID);
CREATE TABLE EMPLEE(EMPID INT,NAME VARCHAR(8),COMPID INT,DESIGNATION VARCHAR(15),SALARY NUMERIC(8),CITY VARCHAR(9),CONSTRAINT EMP PRIMARY KEY(EMPID),CONSTRAINT COMP FOREIGN KEY(COMPID) REFERENCES COMPANY(COMPID));
CREATE TABLE COMPANY(COMPID INT,NAME VARCHAR(6),CITY VARCHAR(9),NO_OF_EMPLOYEE NUMERIC(8),CONSTRAINT CMP PRIMARY KEY(COMPID));

CREATE TABLE STUDNT(STDID INT,NAME VARCHAR(80),CLASS VARCHAR(60),CITY VARCHAR(10),TOTAL_MARKS INT,PERCENTAGE NUMBERIC(5),
DEPTID INT,CONSTRAINT pKEY PRIMARY KEY(STDID),CONSTRAINT DEL FOREIGN KEY(DEPTID)REFERENCES DEPT(DEPTID));


CREATE TABLE STDN(STUDID INT,NAME VARCHAR(80),CLASS VARCHAR(60),CITY VARCHAR(10),TOTAL_MARKS INT,PERCENTAGE NUMERIC(5),
DEPTID INT,CONSTRAINT pKEY PRIMARY KEY(STUDID),CONSTRAINT DEL FOREIGN KEY(DEPTID) REFERENCES DEPT(DEPTID));

CREATE TABLE STAFFS(STAFFID INT,NAME VARCHAR(70),DEPTID INT,DESIGNATION VARCHAR(15),SALARY NUMERIC(8),CITY VARCHAR(80),
CONSTRAINT STF PRIMARY KEY(STAFFID),
CONSTRAINT DEPT FOREIGN KEY(DEPTID) REFERENCES DEPT(DEPTID));
